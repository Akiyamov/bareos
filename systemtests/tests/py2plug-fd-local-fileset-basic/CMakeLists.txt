  create_systemtest(${SYSTEMTEST_PREFIX} ${BASENAME} DISABLED)
endif()

  )
  set_tests_properties(
    "${SYSTEMTEST_PREFIX}py3plug-fd-local-fileset-basic:io_in_core"
    PROPERTIES LABELS "broken"
  )
else()
  create_systemtest(${SYSTEMTEST_PREFIX} ${BASENAME} DISABLED)
endif()
